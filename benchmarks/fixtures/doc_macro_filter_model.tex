\subsection{State equations}

\begin{assumption}[Macro filter dimensions]
\label{assump:macro-filter-dimensions}
Matrices $\trans$ and $\obs$ are conformable, and $\cov$ is symmetric positive
definite whenever the likelihood contribution is evaluated.
\end{assumption}

\begin{equation}
\label{eq:macro-filter-transition}
\state = \trans x_{t-1} + \eta_t
\end{equation}

\begin{align}
\label{eq:macro-filter-innovation}
\innov &= \meas - \obs \state \\
\cov &= \obs P_t \obs^\top + R_t
\end{align}

\begin{equation}
\label{eq:macro-filter-likelihood}
\ell_t = -\frac{1}{2}\log\det\cov - \frac{1}{2}\innov^\top \cov^{-1}\innov
\end{equation}

\begin{proposition}[Repeated notation review]
\label{prop:macro-filter-repeat-notation}
The symbols $\innov$ and $\cov$ are reused in prediction, update, and likelihood
paragraphs; an automated audit should preserve label provenance rather than merge
the occurrences into one generated identifier.
\end{proposition}
